Installation
------------

## CocoaPods
```
platform :ios, '9.0'

target 'TargetName' do
	use_frameworks!
	pod 'devtodev'
end
```

```
target ‘ExtensionTargetName’ do
	platform :ios, ’10.0’
	pod ‘devtodev/AppExtension’
end
```
